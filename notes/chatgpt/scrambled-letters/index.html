The purpose of this test is to test ChatGPT's ability to understand as well as produce text in a jumbled manner. This means it has to first generate the abstract concept of language and then do post-processing to jumble the orthography (but not across word boundaries).

My questions are generated using the following script:

function shuffle(array) {
  for (let cur = array.length - 1; cur >= 0; cur--) {
    if (!/\w/.test(array[cur])) continue;
    const target = Math.floor(Math.random() * (cur + 1));
    let temp = array[cur];
    array[cur] = array[target];
    array[target] = temp;
  }
  return array;
}

function scrambled(text) {
  return text
    .split(" ")
    .map((word) => shuffle(word.split("")).join(""))
    .join(" ");
}
